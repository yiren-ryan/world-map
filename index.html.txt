<!DOCTYPE html>
<html lang="en">
<head>
  <meta charset="utf-8" />
  <title>Interactive World Map</title>
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <link rel="stylesheet" href="https://unpkg.com/leaflet@1.9.3/dist/leaflet.css" />
  <style>
    #map { height: 100vh; }
    .note-popup input, .note-popup textarea {
      width: 100%;
      margin-bottom: 5px;
    }
  </style>
</head>
<body>
  <div id="map"></div>
  <script src="https://unpkg.com/leaflet@1.9.3/dist/leaflet.js"></script>
  <script>
    const map = L.map('map').setView([20, 0], 2);
    L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {
      maxZoom: 5,
      attribution: '&copy; OpenStreetMap contributors'
    }).addTo(map);
    map.on('dblclick', function(e) {
      const popupContent = document.createElement('div');
      popupContent.className = 'note-popup';
      const nameInput = document.createElement('input');
      nameInput.placeholder = 'Your name';
      popupContent.appendChild(nameInput);
      const messageInput = document.createElement('textarea');
      messageInput.placeholder = 'Your message';
      popupContent.appendChild(messageInput);
      const saveBtn = document.createElement('button');
      saveBtn.innerText = 'Save';
      popupContent.appendChild(saveBtn);
      const popup = L.popup()
        .setLatLng(e.latlng)
        .setContent(popupContent)
        .openOn(map);
      saveBtn.onclick = () => {
        const name = nameInput.value;
        const msg = messageInput.value;
        if (name && msg) {
          L.marker(e.latlng)
            .addTo(map)
            .bindPopup(`<strong>${name}</strong><br>${msg}`);
          map.closePopup();
        }
      };
    });
  </script>
</body>
</html>
